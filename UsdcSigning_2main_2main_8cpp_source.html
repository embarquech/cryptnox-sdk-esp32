107static void parse_address(const char *hex, uint8_t out[20])
108{
109 const char *p = hex;
110 if ((p[0] == '0') && ((p[1] == 'x') || (p[1] == 'X'))) {
111 p += 2;
112 }
113 (void)memset(out, 0, 20U);
114 size_t i;
115 for (i = 0U; (i < 20U) && (p[0] != '\0') && (p[1] != '\0'); i++) {
116 uint8_t hi = (uint8_t)((*p >= 'a') ? (*p - 'a' + 10) :
117 (*p >= 'A') ? (*p - 'A' + 10) : (*p - '0'));
118 p++;
119 uint8_t lo = (uint8_t)((*p >= 'a') ? (*p - 'a' + 10) :
120 (*p >= 'A') ? (*p - 'A' + 10) : (*p - '0'));
121 p++;
122 out[i] = (uint8_t)((hi << 4U) | lo);
123 }
124}

139static void build_usdc_calldata(uint8_t out[68], const char *to_hex, uint64_t amount)
140{
141 (void)memset(out, 0, 68U);
142
143 /* Function selector */
144 (void)memcpy(out, TRANSFER_SELECTOR, 4U);
145
146 /* 'to' address: right-aligned in 32-byte slot starting at offset 4 */
147 uint8_t addr[20];
148 parse_address(to_hex, addr);
149 (void)memcpy(out + 4U + 12U, addr, 20U);
150
151 /* amount: right-aligned uint256, 64-bit value fits in the last 8 bytes */
152 size_t j;
153 for (j = 0U; j < 8U; j++) {
154 out[67U - j] = (uint8_t)((amount >> (8U * j)) & 0xFFU);
155 }
156}

172static void signing_loop(CryptnoxWallet &wallet)
173{
174 /* CARD_PIN is a string literal ("000000000"); copy into the pin array. */
175 uint8_t card_pin[CW_MAX_PIN_LENGTH];
176 (void)memset(card_pin, 0U, sizeof(card_pin));
177 (void)memcpy(card_pin, CARD_PIN,
178 (CARD_PIN_LEN < CW_MAX_PIN_LENGTH) ? CARD_PIN_LEN : CW_MAX_PIN_LENGTH);
179
180 uint8_t calldata[68];
181 build_usdc_calldata(calldata, "0x" ADDR_TO, AMOUNT_USDC);
182
183 while (true) {
184 /* ── 1. Wait for card ──────────────────────────────────── */
185 ESP_LOGI(TAG, "Hold Cryptnox card to reader to sign...");
186
187 CW_SecureSession session;
188 bool connected = wallet.connect(session);
189 if (!connected) {
190 vTaskDelay(pdMS_TO_TICKS(100U));
191 continue;
192 }
193
194 /* ── 2. Get fresh nonce (card is present — fetch now) ──── */
195 uint64_t nonce = 0U;
196 if (!eth_rpc_get_nonce(&nonce)) {
197 ESP_LOGE(TAG, "Failed to get nonce — retrying in 5 s");
198 wallet.disconnect(session);
199 vTaskDelay(pdMS_TO_TICKS(5000U));
200 continue;
201 }
202
203 /* ── 3. Build tx ───────────────────────────────────────── */
204 eth_tx_t tx;
205 (void)memset(&tx, 0, sizeof(tx));
206 tx.chain_id = CHAIN_ID_SEPOLIA;
207 tx.nonce = nonce;
208 tx.max_priority_fee = MAX_PRIORITY_FEE;
209 tx.max_fee = MAX_FEE;
210 tx.gas_limit = GAS_LIMIT_ERC20;
211 tx.eth_value = 0U;
212 tx.calldata = calldata;
213 tx.calldata_len = sizeof(calldata);
214 parse_address("0x" ADDR_USDC, tx.to);
215
216 /* ── 4. Encode unsigned tx and hash it ─────────────────── */
217 uint8_t unsigned_tx[TX_BUF_SIZE];
218 size_t unsigned_len = eth_rlp_encode_unsigned(&tx, unsigned_tx, sizeof(unsigned_tx));
219 if (unsigned_len == 0U) {
220 ESP_LOGE(TAG, "RLP encode unsigned overflow");
221 wallet.disconnect(session);
222 vTaskDelay(pdMS_TO_TICKS(2000U));
223 continue;
224 }
225
226 uint8_t hash[CW_HASH_SIZE];
227 keccak256(unsigned_tx, unsigned_len, hash);
228
229 ESP_LOGI(TAG, "Hash to sign:");
230 ESP_LOG_BUFFER_HEX_LEVEL(TAG, hash, CW_HASH_SIZE, ESP_LOG_INFO);
231
232 /* BIP32 Ethereum derivation path: m/44'/60'/0'/0/0
233 * Each level is a 4-byte big-endian uint32; hardened levels have the
234 * high bit set. */
235 static const uint8_t eth_path[20] = {
236 0x80U, 0x00U, 0x00U, 0x2CU, /* 44' */
237 0x80U, 0x00U, 0x00U, 0x3CU, /* 60' */
238 0x80U, 0x00U, 0x00U, 0x00U, /* 0' */
239 0x00U, 0x00U, 0x00U, 0x00U, /* 0 */
240 0x00U, 0x00U, 0x00U, 0x00U, /* 0 */
241 };
242
243 CW_SignRequest req(session,
244 CW_SIGN_DERIVE_K1,
245 CW_SIGN_SIG_ECDSA_LOW_S,
246 CW_SIGN_WITH_PIN);
247 req.hash = hash;
248 req.hashLength = static_cast<uint8_t>(CW_HASH_SIZE);
249 req.derivePath = eth_path;
250 req.derivePathLength = static_cast<uint8_t>(sizeof(eth_path));
251 (void)memcpy(req.pin, card_pin, CW_MAX_PIN_LENGTH);
252
253 CW_SignResult result = wallet.sign(req);
254 wallet.disconnect(session);
255
256 if (result.errorCode != CW_OK) {
257 ESP_LOGE(TAG, "Sign failed: 0x%02X",
258 static_cast<unsigned int>(result.errorCode));
259 vTaskDelay(pdMS_TO_TICKS(2000U));
260 continue;
261 }
262
263 uint8_t sig_r[CW_HASH_SIZE];
264 uint8_t sig_s[CW_HASH_SIZE];
265 (void)memcpy(sig_r, result.signature + CW_SIG_R_OFFSET, CW_HASH_SIZE);
266 (void)memcpy(sig_s, result.signature + CW_SIG_S_OFFSET, CW_HASH_SIZE);
267
268 ESP_LOGI(TAG, "r:");
269 ESP_LOG_BUFFER_HEX_LEVEL(TAG, sig_r, CW_HASH_SIZE, ESP_LOG_INFO);
270 ESP_LOGI(TAG, "s:");
271 ESP_LOG_BUFFER_HEX_LEVEL(TAG, sig_s, CW_HASH_SIZE, ESP_LOG_INFO);
272
273 /* ── 6. Determine v parity via ecrecover ───────────────── */
274 uint8_t v = eth_rpc_ecrecover_parity(hash, sig_r, sig_s);
275 ESP_LOGI(TAG, "v = %u", static_cast<unsigned int>(v));
276
277 /* ── 7. Encode signed tx ───────────────────────────────── */
278 uint8_t signed_tx[TX_BUF_SIZE];
279 size_t signed_len = eth_rlp_encode_signed(&tx, v, sig_r, sig_s,
280 signed_tx, sizeof(signed_tx));
281 if (signed_len == 0U) {
282 ESP_LOGE(TAG, "RLP encode signed overflow");
283 vTaskDelay(pdMS_TO_TICKS(2000U));
284 continue;
285 }
286
287 ESP_LOGI(TAG, "Signed tx (%u bytes):", (unsigned int)signed_len);
288 ESP_LOG_BUFFER_HEX_LEVEL(TAG, signed_tx, signed_len, ESP_LOG_INFO);
289
290 /* ── 8. Broadcast ──────────────────────────────────────── */
291 char tx_hash[68] = { 0 };
292 if (eth_rpc_send_raw_tx(signed_tx, signed_len,
293 tx_hash, sizeof(tx_hash))) {
294 ESP_LOGI(TAG, "TX broadcast OK: %s", tx_hash);
295 } else {
296 ESP_LOGE(TAG, "TX broadcast failed");
297 }
298
299 /* Wait before next iteration so nonce advances on-chain. */
300 vTaskDelay(pdMS_TO_TICKS(15000U));
301 }
302}

314extern "C" void app_main(void)
315{
316 /* ── NVS (required by WiFi driver) ────────────────────────── */
317 esp_err_t nvs_ret = nvs_flash_init();
318 if ((nvs_ret == ESP_ERR_NVS_NO_FREE_PAGES) ||
319 (nvs_ret == ESP_ERR_NVS_NEW_VERSION_FOUND)) {
320 ESP_ERROR_CHECK(nvs_flash_erase());
321 nvs_ret = nvs_flash_init();
322 }
323 ESP_ERROR_CHECK(nvs_ret);
324
325 /* ── PN532 NFC reader (transport selected at the top of this file) ── */
326 pn532_t nfc;
327 (void)memset(&nfc, 0, sizeof(nfc));
328
329 pn532_config_t nfc_cfg;
330 (void)memset(&nfc_cfg, 0, sizeof(nfc_cfg));
331#if SPI_ENABLED
332 spi_bus_config_t buscfg;
333 (void)memset(&buscfg, 0, sizeof(buscfg));
334 buscfg.mosi_io_num = SPI_MOSI;
335 buscfg.miso_io_num = SPI_MISO;
336 buscfg.sclk_io_num = SPI_SCLK;
337 buscfg.quadwp_io_num = SPI_PIN_UNUSED;
338 buscfg.quadhd_io_num = SPI_PIN_UNUSED;
339 buscfg.max_transfer_sz = SPI_MAX_TRANSFER_SZ;
340 ESP_ERROR_CHECK(spi_bus_initialize(SPI2_HOST, &buscfg, SPI_DMA_CH_AUTO));
341
342 nfc_cfg.transport = PN532_TRANSPORT_SPI;
343 nfc_cfg.spi_host = SPI2_HOST;
344 nfc_cfg.pin_cs = NFC_CS;
345 nfc_cfg.skip_bus_init = true;
346#endif
347
348#if I2C_ENABLED
349 nfc_cfg.transport = PN532_TRANSPORT_I2C;
350 nfc_cfg.i2c_port = PN532_I2C_PORT;
351 nfc_cfg.pin_sda = PN532_SDA;
352 nfc_cfg.pin_scl = PN532_SCL;
353 nfc_cfg.pin_irq = PN532_IRQ;
354 nfc_cfg.pin_rst = PN532_RST;
355 nfc_cfg.i2c_clock_hz = PN532_I2C_HZ;
356#endif
357
358 esp_err_t nfc_ret = pn532_init(&nfc, &nfc_cfg);
359 if (nfc_ret != ESP_OK) {
360 ESP_LOGE(TAG, "PN532 init failed");
361 return;
362 }
363
364 /* ── Wallet setup ──────────────────────────────────────────── */
365 ESP32Logger logger;
366 (void)logger.begin(115200UL);
367
368 ESP32CryptoProvider cryptoProvider;
369 ESP32Platform platform;
370 Pn532NfcTransport nfcTransport(&nfc, logger);
371 CryptnoxWallet wallet(nfcTransport, logger, cryptoProvider, platform);
372
373 if (!wallet.begin()) {
374 ESP_LOGE(TAG, "Wallet begin (SAMConfig) failed");
375 return;
376 }
377 /* wallet.begin() already prints the PN532 firmware version internally. */
378
379 /* ── WiFi + RPC ────────────────────────────────────────────── */
380 eth_rpc_init(RPC_URL, "0x" ADDR_FROM);
381#if defined(RPC_PROJECT_ID) && defined(RPC_API_SECRET)
382 eth_rpc_set_auth(RPC_PROJECT_ID, RPC_API_SECRET);
383#endif
384
385 if (!eth_rpc_wifi_connect(WIFI_SSID, WIFI_PASSWORD)) {
386 ESP_LOGE(TAG, "WiFi connect failed — check config.h credentials");
387 return;
388 }
389
390 ESP_LOGI(TAG, "Ready — will sign USDC transfer each card tap");
391
392 signing_loop(wallet);
393}
