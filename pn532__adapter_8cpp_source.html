86bool PN532Adapter::printFirmwareVersion()
87{
88 uint32_t version = 0U;
89 bool result = false;
90
91 if (_initialized) {
92 version = pn532_get_firmware_version(&_dev);
93 if (version == 0U) {
94 _logger.println("PN532 firmware query failed");
95 } else {
96 _logger.print("PN5");
97 _logger.print(static_cast<uint8_t>((version >> FW_IC_SHIFT) & FW_BYTE_MASK), HEX);
98 _logger.print(" Firmware v");
99 _logger.print(static_cast<uint8_t>((version >> FW_VER_SHIFT) & FW_BYTE_MASK), DEC);
100 _logger.print(".");
101 _logger.println(static_cast<uint8_t>((version >> FW_REV_SHIFT) & FW_BYTE_MASK), DEC);
102 result = true;
103 }
104 }
105
106 return result;
107}
