220static uint8_t spi_read_byte(pn532_t *dev)
221{
222 uint8_t rx = 0U;
223 uint8_t tx = 0x00U;
224 spi_transaction_t t;
225 (void)memset(&t, 0, sizeof(t));
226 t.length = PN532_SPI_BITS;
227 t.rxlength = PN532_SPI_BITS;
228 t.tx_buffer = &tx;
229 t.rx_buffer = &rx;
230 (void)spi_device_transmit(dev->spi, &t);
231 return rx;
232}

249static uint8_t read_ready(pn532_t *dev)
250{
251 uint8_t status;
252
253 if (dev->transport == PN532_TRANSPORT_I2C) {
254 status = i2c_read_ready(dev);
255 } else {
256 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_LOW);
257 vTaskDelay(pdMS_TO_TICKS(PN532_CS_TOGGLE_DELAY_MS));
258 spi_write_byte(dev, PN532_SPI_STATREAD);
259 status = spi_read_byte(dev);
260 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_HIGH);
261 }
262
263 return status;
264}

266static void read_data(pn532_t *dev, uint8_t *buff, uint8_t n)
267{
268 if (dev->transport == PN532_TRANSPORT_I2C) {
269 /* PN532 I2C returns N+1 bytes: byte[0] = RDY (0x01), byte[1..N] = payload. */
270 uint8_t tmp[PN532_I2C_RX_MAX + 1U];
271 uint16_t to_read = (uint16_t)n + 1U;
272 if (to_read > sizeof(tmp)) {
273 to_read = (uint16_t)sizeof(tmp);
274 }
275 (void)i2c_master_receive(dev->i2c_dev, tmp, (size_t)to_read, PN532_I2C_TIMEOUT_MS);
276 size_t payload_len = (size_t)to_read - 1U;
277 (void)memcpy(buff, &tmp[1], payload_len);
278 } else {
279 uint8_t i = 0U;
280 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_LOW);
281 vTaskDelay(pdMS_TO_TICKS(PN532_CS_TOGGLE_DELAY_MS));
282 spi_write_byte(dev, PN532_SPI_DATAREAD);
283 for (i = 0U; i < n; i++) {
284 vTaskDelay(pdMS_TO_TICKS(PN532_BYTE_DELAY_MS));
285 buff[i] = spi_read_byte(dev);
286 }
287 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_HIGH);
288 }
289}

291static bool check_ack(pn532_t *dev)
292{
293 uint8_t ackbuff[PN532_ACK_LEN];
294 bool result;
295 (void)memset(ackbuff, 0, sizeof(ackbuff));
296 read_data(dev, ackbuff, PN532_ACK_LEN);
297 result = pn532_buffer_equal(ackbuff, pn532_ack, PN532_ACK_LEN);
298 return result;
299}

315static void write_command(pn532_t *dev, const uint8_t *cmd, uint8_t cmd_len)
316{
317 uint8_t frame_len = (uint8_t)(cmd_len + PN532_FRAME_TFI_OVERHEAD);
318 uint8_t checksum = (uint8_t)(PN532_PREAMBLE + PN532_PREAMBLE + PN532_STARTCODE2);
319 uint8_t i = 0U;
320
321 if (dev->transport == PN532_TRANSPORT_I2C) {
322 /* Build the full frame in a buffer and send it in one I2C transaction. */
323 uint8_t frame[PN532_I2C_TX_MAX];
324 uint16_t pos = 0U;
325
326 frame[pos] = PN532_PREAMBLE; pos++;
327 frame[pos] = PN532_PREAMBLE; pos++;
328 frame[pos] = PN532_STARTCODE2; pos++;
329 frame[pos] = frame_len; pos++;
330 frame[pos] = (uint8_t)((uint8_t)(~frame_len) + 1U); pos++;
331 frame[pos] = PN532_HOSTTOPN532; pos++;
332 checksum = (uint8_t)(checksum + PN532_HOSTTOPN532);
333
334 for (i = 0U; i < cmd_len; i++) {
335 frame[pos] = cmd[i]; pos++;
336 checksum = (uint8_t)(checksum + cmd[i]);
337 }
338 frame[pos] = (uint8_t)(~checksum); pos++;
339 frame[pos] = PN532_POSTAMBLE; pos++;
340
341 (void)i2c_master_transmit(dev->i2c_dev, frame, (size_t)pos, PN532_I2C_TIMEOUT_MS);
342 } else {
343 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_LOW);
344 vTaskDelay(pdMS_TO_TICKS(PN532_CS_TOGGLE_DELAY_MS));
345 spi_write_byte(dev, PN532_SPI_DATAWRITE);
346
347 spi_write_byte(dev, PN532_PREAMBLE);
348 spi_write_byte(dev, PN532_PREAMBLE);
349 spi_write_byte(dev, PN532_STARTCODE2);
350
351 spi_write_byte(dev, frame_len);
352 spi_write_byte(dev, (uint8_t)((uint8_t)(~frame_len) + 1U));
353
354 spi_write_byte(dev, PN532_HOSTTOPN532);
355 checksum = (uint8_t)(checksum + PN532_HOSTTOPN532);
356
357 for (i = 0U; i < cmd_len; i++) {
358 spi_write_byte(dev, cmd[i]);
359 checksum = (uint8_t)(checksum + cmd[i]);
360 }
361 spi_write_byte(dev, (uint8_t)(~checksum));
362 spi_write_byte(dev, PN532_POSTAMBLE);
363 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_HIGH);
364 }
365}

367static bool send_command_check_ack(pn532_t *dev, const uint8_t *cmd,
368 uint8_t cmd_len, uint16_t timeout)
369{
370 uint16_t timer = 0U;
371 bool timed_out = false;
372 bool result = false;
373
374 ESP_LOGI(PN532_LOG_TAG, "═══ cmd=0x%02X cmd_len=%u ═══", cmd[0], (unsigned)cmd_len);
375 write_command(dev, cmd, cmd_len);
376 ESP_LOGI(PN532_LOG_TAG, "→ polling RDY for ACK...");
377
378 /* Wait until the PN532 signals it is ready to send the ACK frame. */
379 while ((!timed_out) && (read_ready(dev) != PN532_SPI_READY)) {
380 if (timeout != 0U) {
381 timer = (uint16_t)(timer + PN532_POLL_INTERVAL_MS);
382 if (timer > timeout) {
383 timed_out = true;
384 }
385 }
386 if (!timed_out) {
387 vTaskDelay(pdMS_TO_TICKS(PN532_POLL_INTERVAL_MS));
388 }
389 }
390
391 if (timed_out) {
392 ESP_LOGE(PN532_LOG_TAG, "→ TIMEOUT waiting RDY for ACK");
393 } else {
394 ESP_LOGI(PN532_LOG_TAG, "→ RDY after %u ms, reading ACK...", (unsigned)timer);
395 bool ack_ok = check_ack(dev);
396 ESP_LOGI(PN532_LOG_TAG, "→ ACK %s", ack_ok ? "OK" : "MISMATCH");
397
398 if (ack_ok) {
399 timer = 0U;
400 timed_out = false;
401
402 /* Wait until the PN532 signals it is ready to send the response. */
403 while ((!timed_out) && (read_ready(dev) != PN532_SPI_READY)) {
404 if (timeout != 0U) {
405 timer = (uint16_t)(timer + PN532_POLL_INTERVAL_MS);
406 if (timer > timeout) {
407 timed_out = true;
408 }
409 }
410 if (!timed_out) {
411 vTaskDelay(pdMS_TO_TICKS(PN532_POLL_INTERVAL_MS));
412 }
413 }
414
415 if (timed_out) {
416 ESP_LOGE(PN532_LOG_TAG, "→ TIMEOUT waiting RDY for response");
417 } else {
418 ESP_LOGI(PN532_LOG_TAG, "→ response RDY after %u ms", (unsigned)timer);
419 }
420 result = !timed_out;
421 }
422 }
423
424 ESP_LOGI(PN532_LOG_TAG, "═══ result=%s ═══", result ? "OK" : "FAIL");
425 return result;
426}

428/* Read one complete PN532 response frame within a single CS-low window.
429 * Handles both normal frames (5-byte header) and extended frames (8-byte header,
430 * indicated by frame[3]==0xFF and frame[4]==0xFF).
431 * CS is released as soon as the last real frame byte is read, preventing
432 * spurious clocks from corrupting the PN532 SPI state machine. */
433static uint16_t read_data_apdu_frame(pn532_t *dev, uint8_t *buff, uint16_t max_len)
434{
435 uint16_t body_len = 0U;
436 uint16_t total_len = 0U;
437 bool is_extended = false;
438
439 if (dev->transport == PN532_TRANSPORT_I2C) {
440 /* PN532 I²C returns 1 RDY byte + N payload bytes in a single read.
441 * Pull the whole maximum response at once, then parse the frame
442 * header to figure out the actual length. */
443 uint8_t tmp[PN532_EXCHANGE_FRAME_MAX + 1U];
444 uint16_t to_read = max_len + 1U;
445 if (to_read > (uint16_t)sizeof(tmp)) {
446 to_read = (uint16_t)sizeof(tmp);
447 }
448 (void)i2c_master_receive(dev->i2c_dev, tmp, (size_t)to_read,
449 PN532_I2C_TIMEOUT_MS);
450 size_t payload_len = (size_t)to_read - (size_t)1U;
451 (void)memcpy(buff, &tmp[1], payload_len);
452
453 is_extended = ((buff[PN532_EXCHANGE_LEN_OFFSET] == PN532_EXT_FRAME_INDICATOR) &&
454 (buff[PN532_EXCHANGE_LEN_OFFSET + 1U] == PN532_EXT_FRAME_INDICATOR));
455 if (is_extended) {
456 body_len = ((uint16_t)buff[PN532_EXT_FRAME_LENHI_OFFSET] << 8U)
457 | (uint16_t)buff[PN532_EXT_FRAME_LENLO_OFFSET];
458 total_len = (uint16_t)PN532_EXT_FRAME_HDR_LEN
459 + body_len
460 + (uint16_t)PN532_FRAME_TAIL_LEN;
461 } else {
462 body_len = (uint16_t)buff[PN532_EXCHANGE_LEN_OFFSET];
463 total_len = (uint16_t)PN532_FRAME_HDR_LEN
464 + body_len
465 + (uint16_t)PN532_FRAME_TAIL_LEN;
466 }
467 if (total_len > max_len) {
468 total_len = max_len;
469 }
470 } else {
471 /* SPI path: byte-by-byte over a single CS-low window. */
472 uint16_t i;
473 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_LOW);
474 vTaskDelay(pdMS_TO_TICKS(PN532_CS_TOGGLE_DELAY_MS));
475 spi_write_byte(dev, PN532_SPI_DATAREAD);
476
477 for (i = 0U; i < PN532_FRAME_HDR_LEN; i++) {
478 vTaskDelay(pdMS_TO_TICKS(PN532_BYTE_DELAY_MS));
479 buff[i] = spi_read_byte(dev);
480 }
481
482 is_extended = ((buff[PN532_EXCHANGE_LEN_OFFSET] == PN532_EXT_FRAME_INDICATOR) &&
483 (buff[PN532_EXCHANGE_LEN_OFFSET + 1U] == PN532_EXT_FRAME_INDICATOR));
484
485 if (is_extended) {
486 /* Read the 3 additional extended-frame header bytes: LEN_H, LEN_L, LCS. */
487 for (i = PN532_FRAME_HDR_LEN; i < PN532_EXT_FRAME_HDR_LEN; i++) {
488 vTaskDelay(pdMS_TO_TICKS(PN532_BYTE_DELAY_MS));
489 buff[i] = spi_read_byte(dev);
490 }
491 body_len = ((uint16_t)buff[PN532_EXT_FRAME_LENHI_OFFSET] << 8U)
492 | (uint16_t)buff[PN532_EXT_FRAME_LENLO_OFFSET];
493 total_len = (uint16_t)PN532_EXT_FRAME_HDR_LEN
494 + body_len
495 + (uint16_t)PN532_FRAME_TAIL_LEN;
496 } else {
497 body_len = (uint16_t)buff[PN532_EXCHANGE_LEN_OFFSET];
498 total_len = (uint16_t)PN532_FRAME_HDR_LEN
499 + body_len
500 + (uint16_t)PN532_FRAME_TAIL_LEN;
501 }
502
503 if (total_len > max_len) {
504 total_len = max_len;
505 }
506
507 i = (uint16_t)(is_extended ? PN532_EXT_FRAME_HDR_LEN : PN532_FRAME_HDR_LEN);
508 while (i < total_len) {
509 vTaskDelay(pdMS_TO_TICKS(PN532_BYTE_DELAY_MS));
510 buff[i] = spi_read_byte(dev);
511 i++;
512 }
513
514 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_HIGH);
515 }
516
517 return total_len;
518}

524static esp_err_t pn532_init_spi(pn532_t *dev, const pn532_config_t *config)
525{
526 gpio_config_t io_conf;
527 spi_bus_config_t buscfg;
528 spi_device_interface_config_t devcfg;
529 esp_err_t ret;
530
531 (void)memset(&io_conf, 0, sizeof(io_conf));
532 io_conf.pin_bit_mask = (GPIO_PIN_BITMASK_BASE << (uint32_t)config->pin_cs);
533 io_conf.mode = GPIO_MODE_OUTPUT;
534 io_conf.pull_up_en = GPIO_PULLUP_DISABLE;
535 io_conf.pull_down_en = GPIO_PULLDOWN_DISABLE;
536 io_conf.intr_type = GPIO_INTR_DISABLE;
537 (void)gpio_config(&io_conf);
538 (void)gpio_set_level(config->pin_cs, GPIO_LEVEL_HIGH);
539 dev->pin_cs = config->pin_cs;
540
541 if (config->skip_bus_init) {
542 ret = ESP_OK;
543 } else {
544 (void)memset(&buscfg, 0, sizeof(buscfg));
545 buscfg.mosi_io_num = config->pin_mosi;
546 buscfg.miso_io_num = config->pin_miso;
547 buscfg.sclk_io_num = config->pin_sclk;
548 buscfg.quadwp_io_num = PN532_SPI_NO_PIN;
549 buscfg.quadhd_io_num = PN532_SPI_NO_PIN;
550 buscfg.max_transfer_sz = 0;
551 ret = spi_bus_initialize(config->spi_host, &buscfg, SPI_DMA_CH_AUTO);
552 }
553
554 if (ret == ESP_OK) {
555 (void)memset(&devcfg, 0, sizeof(devcfg));
556 devcfg.mode = PN532_SPI_MODE;
557 devcfg.clock_speed_hz = (int)PN532_SPI_CLOCK_HZ;
558 devcfg.spics_io_num = PN532_SPI_NO_PIN;
559 devcfg.queue_size = PN532_SPI_QUEUE_SIZE;
560 devcfg.flags = SPI_DEVICE_BIT_LSBFIRST;
561 ret = spi_bus_add_device(config->spi_host, &devcfg, &dev->spi);
562
563 if ((ret != ESP_OK) && (!config->skip_bus_init)) {
564 (void)spi_bus_free(config->spi_host);
565 }
566 }
567
568 if (ret == ESP_OK) {
569 /* Wake-up sequence: send 0x55 0x55 + 3x preamble while CS is low. */
570 (void)gpio_set_level(dev->pin_cs, GPIO_LEVEL_LOW);
571 vTaskDelay(pdMS_TO_TICKS(PN532_CS_TOGGLE_DELAY_MS));

584static esp_err_t pn532_init_i2c(pn532_t *dev, const pn532_config_t *config)
585{
586 esp_err_t ret;
587
588 /* Optional reset pulse. */
589 if (config->pin_rst >= 0) {
590 gpio_config_t rst_cfg;
591 (void)memset(&rst_cfg, 0, sizeof(rst_cfg));
592 rst_cfg.pin_bit_mask = (GPIO_PIN_BITMASK_BASE << (uint32_t)config->pin_rst);
593 rst_cfg.mode = GPIO_MODE_OUTPUT;
594 (void)gpio_config(&rst_cfg);
595 (void)gpio_set_level(config->pin_rst, GPIO_LEVEL_LOW);
596 vTaskDelay(pdMS_TO_TICKS(50));
597 (void)gpio_set_level(config->pin_rst, GPIO_LEVEL_HIGH);
598 vTaskDelay(pdMS_TO_TICKS(100));
599 dev->pin_rst = config->pin_rst;
600 } else {
601 dev->pin_rst = -1;
602 }
603
604 /* Optional IRQ input (configured but polling is used). */
605 if (config->pin_irq >= 0) {
606 gpio_config_t irq_cfg;
607 (void)memset(&irq_cfg, 0, sizeof(irq_cfg));
608 irq_cfg.pin_bit_mask = (GPIO_PIN_BITMASK_BASE << (uint32_t)config->pin_irq);
609 irq_cfg.mode = GPIO_MODE_INPUT;
610 irq_cfg.pull_up_en = GPIO_PULLUP_ENABLE;
611 (void)gpio_config(&irq_cfg);
612 dev->pin_irq = config->pin_irq;
613 } else {
614 dev->pin_irq = -1;
615 }
616
617 /* Create the I2C master bus (new IDF v5.x API). */
618 i2c_master_bus_config_t bus_cfg;
619 (void)memset(&bus_cfg, 0, sizeof(bus_cfg));
620 bus_cfg.i2c_port = config->i2c_port;
621 bus_cfg.sda_io_num = config->pin_sda;
622 bus_cfg.scl_io_num = config->pin_scl;
623 bus_cfg.clk_source = I2C_CLK_SRC_DEFAULT;
624 bus_cfg.glitch_ignore_cnt = 7;
625 bus_cfg.flags.enable_internal_pullup = true;
626
627 ret = i2c_new_master_bus(&bus_cfg, &dev->i2c_bus);
628 if (ret != ESP_OK) {
629 ESP_LOGE(PN532_LOG_TAG, "i2c_new_master_bus failed: %d", ret);
630 } else {
631 /* Attach the PN532 device on the bus. */
632 i2c_device_config_t dev_cfg;
633 (void)memset(&dev_cfg, 0, sizeof(dev_cfg));
634 dev_cfg.dev_addr_length = I2C_ADDR_BIT_LEN_7;
635 dev_cfg.device_address = PN532_I2C_ADDRESS;
636 dev_cfg.scl_speed_hz = (config->i2c_clock_hz != 0U) ? config->i2c_clock_hz : 100000U;
637
638 ret = i2c_master_bus_add_device(dev->i2c_bus, &dev_cfg, &dev->i2c_dev);
639 if (ret != ESP_OK) {
640 ESP_LOGE(PN532_LOG_TAG, "i2c_master_bus_add_device failed: %d", ret);
641 (void)i2c_del_master_bus(dev->i2c_bus);
642 dev->i2c_bus = NULL;
643 } else {
644 /* Adafruit_PN532::begin does a small delay between bus init
645 * and the first I2C transaction; mirror it. */
646 vTaskDelay(pdMS_TO_TICKS(10));
647
648 ESP_LOGI(PN532_LOG_TAG,
649 "I2C master bus installed on port %d (SDA=%d SCL=%d %lu Hz)",
650 config->i2c_port, (int)config->pin_sda, (int)config->pin_scl,
651 (unsigned long)dev_cfg.scl_speed_hz);
652 }
653 }
654
655 return ret;
656}

658/******************************************************************
659 * Public API
660 ******************************************************************/
661
662esp_err_t pn532_init(pn532_t *dev, const pn532_config_t *config)
663{
664 esp_err_t ret;
665
666 (void)memset(dev, 0, sizeof(*dev));
667 dev->transport = config->transport;
668
669 if (config->transport == PN532_TRANSPORT_I2C) {
670 ret = pn532_init_i2c(dev, config);
671 } else {
672 ret = pn532_init_spi(dev, config);
673 }
674
675 if (ret == ESP_OK) {
676 /* Wake-up sequence per PN532 datasheet section 8.3.2.6 + Adafruit
677 * pattern. When the host (ESP32) soft-reboots with the PN532 still
678 * powered, the chip is in Soft-Power-Down mode where:
679 *
680 * 1. The main I2C peripheral is OFF
681 * 2. A dedicated "wake-up watcher" circuit listens on the bus
682 * 3. Any I2C transaction matching the slave address triggers it
683 * 4. The firmware must then identify the wake-up source AND
684 * explicitly disable the watcher AND enable the main I2C
685 *
686 * Step 4 takes time — and the first transaction that triggers the
687 * wake-up is LOST during the switch. So we send a sacrificial
688 * "trigger" frame first, wait long enough for the firmware to enable
689 * the main I2C, then proceed with the real commands. */
690 if (config->transport == PN532_TRANSPORT_I2C) {
691 /* Trigger frame — full SAMConfig payload sent and discarded. We
692 * ignore the result because the chip is still waking up. */
693 ESP_LOGI(PN532_LOG_TAG, "I2C wake-up trigger (sacrificial SAMConfig)");
694 (void)pn532_sam_config(dev);
695
696 /* Give the firmware plenty of time to disable the wake-up block
697 * and enable the main I2C peripheral. */
698 vTaskDelay(pdMS_TO_TICKS(500));
699
700 /* Now the real init sequence — SAMConfig again to put the chip
701 * in Normal Mode with the parameters we want. */
702 ESP_LOGI(PN532_LOG_TAG, "Real SAMConfig after wake-up");
703 (void)pn532_sam_config(dev);
704 vTaskDelay(pdMS_TO_TICKS(PN532_SYNC_DELAY_MS));
705 }
706
707 /* Use pn532_get_firmware_version (not just send_command_check_ack)
708 * because it actually READS the response payload — draining the
709 * PN532's output FIFO. If we leave the response sitting in the
710 * FIFO, the next command's read transactions get the stale firmware
711 * bytes instead of the new ACK, leading to mysterious timeouts in
712 * the wallet's SAMConfig that follows our init. */
713 (void)pn532_get_firmware_version(dev);
714 vTaskDelay(pdMS_TO_TICKS(PN532_SYNC_DELAY_MS));
715
716 ESP_LOGI(PN532_LOG_TAG, "PN532 initialized (%s)",
717 (config->transport == PN532_TRANSPORT_I2C) ? "I2C" : "SPI");
718 }
719
720 return ret;
721}

723uint32_t pn532_get_firmware_version(pn532_t *dev)
724{
725 uint8_t pn532_packetbuffer[PN532_FIRMWARE_RESP_LEN];
726 uint32_t response = 0U;
727 bool ack_received = false;
728
729 (void)memset(pn532_packetbuffer, 0, sizeof(pn532_packetbuffer));
730 pn532_packetbuffer[0] = PN532_FIRMWAREVERSION;
731 ack_received = send_command_check_ack(dev, pn532_packetbuffer,
732 PN532_FIRMWARE_CMD_LEN, PN532_CMD_TIMEOUT_MS);
733
734 if (!ack_received) {
735 ESP_LOGE(PN532_LOG_TAG, "No ACK from PN532");
736 } else {
737 read_data(dev, pn532_packetbuffer, PN532_FIRMWARE_RESP_LEN);
738 ESP_LOG_BUFFER_HEX_LEVEL(PN532_LOG_TAG, pn532_packetbuffer,
739 PN532_FIRMWARE_RESP_LEN, ESP_LOG_INFO);
740
741 if (!pn532_buffer_equal(pn532_packetbuffer, pn532_response_fw, PN532_FIRMWARE_HDR_LEN)) {
742 ESP_LOGE(PN532_LOG_TAG, "Unexpected firmware response");
743 } else {
744 response = (uint32_t)pn532_packetbuffer[PN532_FW_IC_OFFSET];
745 response <<= PN532_BYTE_SHIFT_BITS;
746 response |= (uint32_t)pn532_packetbuffer[PN532_FW_VER_OFFSET];
747 response <<= PN532_BYTE_SHIFT_BITS;
748 response |= (uint32_t)pn532_packetbuffer[PN532_FW_REV_OFFSET];
749 response <<= PN532_BYTE_SHIFT_BITS;
750 response |= (uint32_t)pn532_packetbuffer[PN532_FW_SUPPORT_OFFSET];
751 }
752 }
753
754 return response;
755}

810bool pn532_send_apdu(pn532_t *dev, const uint8_t *apdu, uint8_t apdu_len,
811 uint8_t *response, uint16_t *response_len)
812{
813 uint8_t cmd[PN532_MAX_APDU_LEN + PN532_EXCHANGE_CMD_OVERHEAD];
814 uint8_t frame[PN532_EXCHANGE_FRAME_MAX];
815 bool result = false;
816
817 (void)memset(cmd, 0, sizeof(cmd));
818 (void)memset(frame, 0, sizeof(frame));
819
820 if (apdu_len <= PN532_MAX_APDU_LEN) {
821 uint8_t cmd_total_len = 0U;
822 bool ack_received = false;
823
824 cmd[0] = PN532_INDATAEXCHANGE;
825 cmd[1] = PN532_EXCHANGE_TG;
826 (void)memcpy(&cmd[PN532_EXCHANGE_CMD_OVERHEAD], apdu, apdu_len);
827 cmd_total_len = (uint8_t)(apdu_len + PN532_EXCHANGE_CMD_OVERHEAD);
828
829 ack_received = send_command_check_ack(dev, cmd, cmd_total_len, PN532_APDU_TIMEOUT_MS);
830
831 if (ack_received) {
832 (void)read_data_apdu_frame(dev, frame, (uint16_t)PN532_EXCHANGE_FRAME_MAX);
833
834 bool is_extended = ((frame[PN532_EXCHANGE_LEN_OFFSET] == PN532_EXT_FRAME_INDICATOR) &&
835 (frame[PN532_EXCHANGE_LEN_OFFSET + 1U] == PN532_EXT_FRAME_INDICATOR));
836
837 uint8_t err_byte = 0U;
838 uint16_t data_offset = 0U;
839 uint16_t data_len = 0U;
840
841 if (is_extended) {
842 uint16_t body_len = ((uint16_t)frame[PN532_EXT_FRAME_LENHI_OFFSET] << 8U)
843 | (uint16_t)frame[PN532_EXT_FRAME_LENLO_OFFSET];
844 err_byte = frame[PN532_EXT_EXCHANGE_ERR_OFFSET];
845 data_offset = (uint16_t)PN532_EXT_EXCHANGE_DATA_OFFSET;
846 data_len = (body_len >= (uint16_t)PN532_EXCHANGE_LEN_BIAS)
847 ? (body_len - (uint16_t)PN532_EXCHANGE_LEN_BIAS)
848 : 0U;
849 } else {
850 uint8_t len_field = frame[PN532_EXCHANGE_LEN_OFFSET];
851 err_byte = frame[PN532_EXCHANGE_STATUS_OFFSET];
852 data_offset = (uint16_t)PN532_EXCHANGE_DATA_OFFSET;
853 if (len_field >= (uint8_t)PN532_EXCHANGE_LEN_BIAS) {
854 uint8_t raw_data_len = (uint8_t)(len_field - (uint8_t)PN532_EXCHANGE_LEN_BIAS);
855 data_len = (uint16_t)raw_data_len;
856 } else {
857 data_len = 0U;
858 }
859 }
860
861 if (err_byte == PN532_EXCHANGE_STATUS_OK) {
862 uint16_t buf_cap = *response_len;
863 uint16_t copy_len = (data_len <= buf_cap) ? data_len : buf_cap;
864
865 (void)memcpy(response, &frame[data_offset], (size_t)copy_len);
866 *response_len = copy_len;
867 result = true;
868 } else {
869 ESP_LOGE(PN532_LOG_TAG, "APDU INS=0x%02X: ERR=0x%02X",
870 (unsigned)apdu[1], (unsigned)err_byte);
871 }
872 } else {
873 ESP_LOGE(PN532_LOG_TAG, "APDU INS=0x%02X: no ACK", (unsigned)apdu[1]);
874 }
875 }
876
877 return result;
878}
