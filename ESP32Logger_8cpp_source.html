39static uint32_t clamp_base(int base)
40{
41 uint32_t result = static_cast<uint32_t>(DEC);
42 bool in_range = ((static_cast<uint32_t>(base) >= NUM_BASE_MIN) &&
43 (static_cast<uint32_t>(base) <= NUM_BASE_MAX));
44 if (in_range) {
45 result = static_cast<uint32_t>(base);
46 }
47 return result;
48}

58static void write_uint_to_uart(uint32_t value, uint32_t base)
59{
60 char buf[NUM_BUF_SIZE] = { 0 };
61 uint32_t pos = NUM_BUF_SIZE - 1U;
62 uint32_t v = value;
63
64 buf[pos] = '\0';
65
66 if (v == 0U) {
67 pos--;
68 buf[pos] = '0';
69 } else {
70 while (v > 0U) {
71 pos--;
72 buf[pos] = HEX_CHARS[v % base];
73 v = v / base;
74 }
75 }
76
77 (void)fputs(&buf[pos], stdout);
78}

84bool ESP32Logger::begin(unsigned long baudRate)
85{
86 uart_config_t cfg = { 0 };
87 cfg.baud_rate = static_cast<int>(baudRate);
88 cfg.data_bits = UART_DATA_8_BITS;
89 cfg.parity = UART_PARITY_DISABLE;
90 cfg.stop_bits = UART_STOP_BITS_1;
91 cfg.flow_ctrl = UART_HW_FLOWCTRL_DISABLE;
92 cfg.rx_flow_ctrl_thresh = UART_RX_FLOW_THRESH_NONE;
93 cfg.source_clk = UART_SCLK_DEFAULT;
94
95 esp_err_t err = uart_param_config(UART_LOG_PORT, &cfg);
96 m_initialized = (err == ESP_OK);
97
98 return m_initialized;
99}

146void ESP32Logger::print(int value, int base)
147{
148 if (m_initialized) {
149 uint32_t safe_base = clamp_base(base);
150 bool is_negative_decimal = ((value < 0) && (safe_base == static_cast<uint32_t>(DEC)));
151 if (is_negative_decimal) {
152 (void)fputc(static_cast<int>('-'), stdout);
153 /* Two's-complement negation — avoids UB on INT_MIN */
154 uint32_t abs_val = (~static_cast<uint32_t>(value)) + 1U;
155 write_uint_to_uart(abs_val, safe_base);
156 } else {
157 write_uint_to_uart(static_cast<uint32_t>(value), safe_base);
158 }
159 }
160}
