28static void run_connect_loop(CryptnoxWallet &wallet)
29{
30 while (true) {
31 CW_SecureSession session{};
32 bool connected = wallet.connect(session);
33
34 if (connected) {
35 ESP_LOGI(TAG, "Card connected, secure channel established");
36
37 CW_CardInfo info{};
38 bool infoOk = wallet.getCardInfo(session, &info);
39 if (infoOk) {
40 ESP_LOGI(TAG, "Owner name : %s", info.name);
41 ESP_LOGI(TAG, "Owner email: %s", info.email);
42 } else {
43 ESP_LOGW(TAG, "getCardInfo failed");
44 }
45 } else {
46 ESP_LOGW(TAG, "Card not detected or secure channel failed");
47 }
48
49 wallet.disconnect(session);
50 vTaskDelay(pdMS_TO_TICKS(LOOP_DELAY_MS));
51 }
52}

66 pn532_t nfc = {};

69 .pin_cs = NFC_CS,

72 esp_err_t nfc_ret = pn532_init(&nfc, &nfc_cfg);

74 if (nfc_ret == ESP_OK) {
75 ESP32Logger logger;

78 ESP32CryptoProvider cryptoProvider;
